**.github/workflows/ci.yml** -- Triggered on pull_request to main. Two parallel jobs:

```yaml
name: CI

on:
  pull_request:
    branches: [main]

jobs:
  build-and-test-spfx:
    name: Build & Test SPFx
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: spfx
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'
          cache-dependency-path: spfx/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Build and test (production)
        run: npx heft test --clean --production

  build-and-test-api:
    name: Build & Test API
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: api
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-dotnet@v4
        with:
          dotnet-version: '10.0.x'

      - name: Restore dependencies
        run: dotnet restore

      - name: Build
        run: dotnet build --no-restore --configuration Release

      - name: Test
        run: dotnet test tests/WissensHub.Tests --no-build --configuration Release -e AZURE_FUNCTIONS_ENVIRONMENT=Development
```

Key details:
- MUST use `npx heft test --clean --production` (NOT gulp -- per CLAUDE.md memory and Phase 1 decision, SPFx toolchain is Heft-only)
- API tests need `-e AZURE_FUNCTIONS_ENVIRONMENT=Development` for the synthetic dev identity bypass (Phase 4 decision)
- Two parallel jobs for faster feedback
- No Azure login needed for CI (just build + test)
- `cache: 'npm'` with `cache-dependency-path` for faster npm ci

**.github/workflows/cd.yml** -- Triggered on push to main. Five jobs with dependencies:

```yaml
name: CD

on:
  push:
    branches: [main]

permissions:
  id-token: write
  contents: read

jobs:
  build:
    name: Build & Test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      # --- SPFx Build ---
      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'
          cache-dependency-path: spfx/package-lock.json

      - name: Install SPFx dependencies
        working-directory: spfx
        run: npm ci

      - name: Build and test SPFx (production)
        working-directory: spfx
        run: |
          npx heft test --clean --production
          npx heft package-solution --production

      - name: Upload SPFx package
        uses: actions/upload-artifact@v4
        with:
          name: spfx-package
          path: spfx/sharepoint/solution/wissens-hub.sppkg

      # --- API Build ---
      - uses: actions/setup-dotnet@v4
        with:
          dotnet-version: '10.0.x'

      - name: Restore API dependencies
        working-directory: api
        run: dotnet restore

      - name: Build API
        working-directory: api
        run: dotnet build --no-restore --configuration Release

      - name: Test API
        working-directory: api
        run: dotnet test tests/WissensHub.Tests --no-build --configuration Release -e AZURE_FUNCTIONS_ENVIRONMENT=Development

      - name: Publish API
        working-directory: api
        run: dotnet publish src/WissensHub.Functions --configuration Release --output ./publish

      - name: Upload API publish output
        uses: actions/upload-artifact@v4
        with:
          name: api-publish
          path: api/publish/

      # --- Migration Bundle ---
      - name: Build migration bundle
        run: |
          dotnet tool install --global dotnet-ef
          cd api && dotnet ef migrations bundle \
            --project src/WissensHub.Infrastructure \
            --startup-project src/WissensHub.Functions \
            --output ../efbundle \
            --self-contained

      - name: Upload migration bundle
        uses: actions/upload-artifact@v4
        with:
          name: migration-bundle
          path: efbundle

  deploy-infra:
    name: Deploy Infrastructure
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Azure Login (OIDC)
        uses: azure/login@v2
        with:
          client-id: ${{ vars.AZURE_CLIENT_ID }}
          tenant-id: ${{ vars.AZURE_TENANT_ID }}
          subscription-id: ${{ vars.AZURE_SUBSCRIPTION_ID }}

      - name: Deploy Bicep
        run: |
          az deployment group create \
            --resource-group ${{ vars.AZURE_RESOURCE_GROUP }} \
            --template-file infra/main.bicep \
            --parameters infra/parameters/prod.bicepparam \
            --parameters sqlAdminPassword=${{ secrets.SQL_ADMIN_PASSWORD }}

  deploy-migrations:
    name: Run Database Migrations
    needs: [build, deploy-infra]
    runs-on: ubuntu-latest
    steps:
      - name: Download migration bundle
        uses: actions/download-artifact@v4
        with:
          name: migration-bundle

      - name: Azure Login (OIDC)
        uses: azure/login@v2
        with:
          client-id: ${{ vars.AZURE_CLIENT_ID }}
          tenant-id: ${{ vars.AZURE_TENANT_ID }}
          subscription-id: ${{ vars.AZURE_SUBSCRIPTION_ID }}

      - name: Run EF Core migrations
        run: |
          chmod +x ./efbundle
          ./efbundle --connection "${{ secrets.SQL_CONNECTION_STRING }}"

  deploy-functions:
    name: Deploy Azure Functions
    needs: deploy-migrations
    runs-on: ubuntu-latest
    steps:
      - name: Download API publish output
        uses: actions/download-artifact@v4
        with:
          name: api-publish
          path: ./publish

      - name: Azure Login (OIDC)
        uses: azure/login@v2
        with:
          client-id: ${{ vars.AZURE_CLIENT_ID }}
          tenant-id: ${{ vars.AZURE_TENANT_ID }}
          subscription-id: ${{ vars.AZURE_SUBSCRIPTION_ID }}

      - name: Deploy to Azure Functions
        uses: Azure/functions-action@v1
        with:
          app-name: ${{ vars.AZURE_FUNCTION_APP_NAME }}
          package: ./publish

  deploy-spfx:
    name: Deploy SPFx Package
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Download SPFx package
        uses: actions/download-artifact@v4
        with:
          name: spfx-package

      - uses: actions/setup-node@v4
        with:
          node-version: '22'

      - name: Install CLI for Microsoft 365
        run: npm install @pnp/cli-microsoft365@latest -g

      - name: Login to M365 (Federated Identity)
        run: |
          m365 login --authType federated \
            --clientId ${{ vars.M365_CLIENT_ID }} \
            --tenantId ${{ vars.AZURE_TENANT_ID }}

      - name: Deploy to tenant app catalog
        run: |
          m365 spo app add \
            --filePath ./wissens-hub.sppkg \
            --appCatalogUrl ${{ vars.SPO_APP_CATALOG_URL }} \
            --overwrite
          m365 spo app deploy \
            --name wissens-hub.sppkg \
            --appCatalogUrl ${{ vars.SPO_APP_CATALOG_URL }}
```

Key design decisions:
- **Job dependency chain (per D-05, D-06):** build -> deploy-infra -> deploy-migrations -> deploy-functions. This ensures Bicep provisions resources before migrations run, and migrations complete before code deploys.
- **deploy-spfx runs in parallel** with deploy-infra/deploy-migrations/deploy-functions (only needs `build` artifacts, no Azure dependency)
- **OIDC permissions** at workflow level: `id-token: write` + `contents: read` (per D-07, per pitfall #1 in research)
- **Two separate Entra ID apps** for Azure and M365 (per pitfall #7 in research): Azure deployment uses `vars.AZURE_CLIENT_ID`, M365 deployment uses `vars.M365_CLIENT_ID`
- **GitHub vars (not secrets)** for non-sensitive values: client IDs, tenant ID, subscription ID, resource group, function app name, app catalog URL. Only `SQL_CONNECTION_STRING` and `SQL_ADMIN_PASSWORD` are GitHub secrets.
- **Migration bundle** is self-contained (no .NET SDK needed at runtime, per research recommendation)
- **SPFx package name** is `wissens-hub.sppkg` (from `spfx/config/package-solution.json` paths.zippedPackage)
- **No manual approval gates** per D-06 (solo portfolio project)
- **Auto-deploy on merge** per D-06 (branch protection on PRs is the quality gate)
- Uses `npx heft test --clean --production` and `npx heft package-solution --production` (NOT gulp)
