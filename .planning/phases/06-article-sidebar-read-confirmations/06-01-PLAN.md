function formatGermanDate(date: Date): string {
  var day = date.getDate();

**useEffect for heading extraction (mount + delayed retry):**
```typescript
function extractHeadings(): IHeadingItem[] {
  var elements = document.querySelectorAll('.CanvasZone h2, .CanvasZone h3');
  var items: IHeadingItem[] = [];
  for (var i = 0; i < elements.length; i++) {
    var el = elements[i] as HTMLElement;
    if (!el.id) {
      el.id = 'toc-heading-' + i;
    }
    items.push({
      id: el.id,
      text: el.textContent || '',
      level: el.tagName === 'H2' ? 2 : 3,
    });
  }
  return items;
}
```

Run extraction on mount, then retry after 500ms timeout for late-rendering content:
```typescript
React.useEffect(() => {
  var result = extractHeadings();
  if (result.length > 0) {
    setHeadings(result);
    return;
  }
  var timer = setTimeout(function() {
    setHeadings(extractHeadings());
  }, 500);
  return function() { clearTimeout(timer); };
}, []);
```

**useEffect for IntersectionObserver active section highlighting:**
```typescript
React.useEffect(() => {
  if (headings.length === 0) return;

  var observer = new IntersectionObserver(
    function(entries) {
      for (var i = 0; i < entries.length; i++) {
        if (entries[i].isIntersecting) {
          setActiveHeadingId(entries[i].target.id);
        }
      }
    },
    { rootMargin: '-10% 0px -85% 0px', threshold: 0 }
  );

  headings.forEach(function(h) {
    var el = document.getElementById(h.id);
    if (el) observer.observe(el);
  });

  return function() { observer.disconnect(); };
}, [headings]);
```

**Click handler:** `element.scrollIntoView({ behavior: 'smooth', block: 'start' })`.
