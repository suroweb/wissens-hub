**Query hooks pattern:**

Each query hook follows this pattern:
1. Get services from `useWissensHub()`
2. Use `React.useState<QueryState<T>>` initialized to `{ status: 'idle' }`
3. Use `React.useEffect` to trigger fetch on mount (transitions idle -> loading -> success/error)
4. Return `{ state: QueryState<T>, refetch: () => void }`

The `refetch` function re-runs the same fetch logic, resetting to loading first. The useEffect runs the fetch immediately on mount (or when dependencies change).

```typescript
// Template for query hooks:
function useXxxQuery(deps?: any): { state: QueryState<T>; refetch: () => void } {
  const { services } = useWissensHub();
  const [state, setState] = React.useState<QueryState<T>>({ status: 'idle' });

  const fetch = React.useCallback(async () => {
    setState({ status: 'loading' });
    const result = await services.xxxService.method(deps);
    if (result.success) {
      setState({ status: 'success', data: result.data });
    } else {
      setState({ status: 'error', error: result.error });
    }
  }, [services, deps]);

  React.useEffect(() => { fetch(); }, [fetch]);

  return { state, refetch: fetch };
}
```

**Command hooks pattern:**

Each command hook follows this pattern:
1. Get services from `useWissensHub()`
2. Use `React.useState<CommandState>` initialized to `{ status: 'idle' }`
3. Return `{ state: CommandState, execute: (...args) => Promise<boolean> }`
4. The `execute` function transitions idle -> executing -> success/error, returns true on success

```typescript
// Template for command hooks:
function useXxxCommand(): { state: CommandState; execute: (arg: T) => Promise<boolean> } {
  const { services } = useWissensHub();
  const [state, setState] = React.useState<CommandState>({ status: 'idle' });

  const execute = React.useCallback(async (arg: T): Promise<boolean> => {
    setState({ status: 'executing' });
    const result = await services.xxxService.method(arg);
    if (result.success) {
      setState({ status: 'success' });
      return true;
    } else {
      setState({ status: 'error', error: result.error });
      return false;
    }
  }, [services]);

  return { state, execute };
}
```
